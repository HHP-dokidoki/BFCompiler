std::optional<std::vector<Character> > PreProcess(std::ifstream& input)
{
    input.seekg(0, std::ios::beg);
    std::streampos begin = input.tellg();
    input.seekg(0, std::ios::end);
    std::streampos end   = input.tellg();
    input.seekg(0, std::ios::beg);

    size_t file_length = static_cast<size_t>(end - begin);

    std::vector<Character> res;
    res.reserve(file_length + 256);

    char c;
    size_t line = 0;
    size_t col  = 0;
    while (input.good()) {
        c = input.get();
        
        if (c == EOF) {
            break;
        }

        if (c == '\n') {
            line ++;
            col = 0;
            continue;
        } else if (c == '#') {
            // skip this line 
            line ++;
            col = 0;
            input.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        } else if (c == '\r' || c == '\t' || c == ' ') {
            col ++;
            // ignore whitespace
            continue;
        } else {
            col ++;
            res.push_back((Character){c, line, col});
        }
    }
    if (input.bad()) {
            // Add Error Message
        return std::nullopt;
    }
    return res;
}

std::optional<std::vector<Instruction> > Compile(std::vector<Character>& raw_code)
{
    std::vector<Instruction> res;
    res.reserve(raw_code.size() + 256);


    // Also optimize while compiling
    size_t cnt = 1;
    for (auto ptc = raw_code.begin(); ptc != raw_code.end(); ptc++) {
        cnt = 1;
        switch (ptc->c)
        {
            case '>':
                
                while (ptc + 1 != raw_code.end() && (ptc + 1)->c == '>') {
                    cnt ++;
                    ptc ++;
                }
                res.push_back((Instruction){TokenType::RGH, cnt});
                break;
            case '<':

                while (ptc + 1 != raw_code.end() && (ptc + 1)->c == '<') {
                    cnt ++;
                    ptc ++;
                }
                res.push_back((Instruction){TokenType::LEF, cnt});
                break;
            case '+':
                while (ptc + 1 != raw_code.end() && (ptc + 1)->c == '+') {
                    cnt ++;
                    ptc ++;
                }
                res.push_back((Instruction){TokenType::ADD, cnt});
                break;
            case '-':
                while (ptc + 1 != raw_code.end() && (ptc + 1)->c == '-') {
                    cnt ++;
                    ptc ++;
                }
                res.push_back((Instruction){TokenType::SUB, cnt});
                break;
            case '[':
                res.push_back((Instruction){TokenType::LOP, 0});
                break;
            case ']':
                res.push_back((Instruction){TokenType::ROP, 0});
                break;
            case ',':
                res.push_back((Instruction){TokenType::IN, 0});
                break;
            case '.':
                res.push_back((Instruction){TokenType::OUT, 0});
                break;
            default :
                // Log Error Message
                break;
        }
    }

    // for (auto r : res) {
    //     std::cout << s[(int)r.type] << ' ' << r.count << '\n';
    // }

    return res;
}


void Generate_CCode(std::ofstream& target_file, std::vector<Instruction>& Inst_list)
{
    target_file << "#include <stdio.h>\n";
    target_file << "unsigned char array[3000] = {0};\n";
    target_file << "char* ptc = array;\n";
    target_file << "int main()\n{\n";
    for (auto cmd : Inst_list) {
        switch (cmd.type) {
            case TokenType::ADD :
                target_file << "(*ptc) += " << cmd.count << ";\n";
                break;
            case TokenType::SUB :
                target_file << "(*ptc) -= " << cmd.count << ";\n";
                break;
            case TokenType::LEF :
                target_file << "ptc -= " << cmd.count << ";\n";
                break;
            case TokenType::RGH :
                target_file << "ptc += " << cmd.count << ";\n";
                break;
            case TokenType::LOP :
                target_file << "while(*ptc) {\n";
                break;
            case TokenType::ROP :
                target_file << "}\n";
                break;
            case TokenType::OUT :
                target_file << "printf(\"%c\", *ptc);\n";
                break;
            case TokenType::IN :
                target_file << "scanf(\"%c\", ptc);\n";
                break;
        }
    }
    target_file << "while(1){;}\n";
    target_file << "return 0;\n";
    target_file <<"}\n";
}
